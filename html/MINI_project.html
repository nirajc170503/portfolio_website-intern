def strip(data):
    l=[]
    string=''
    for i in data["rate"]:
        if type(i)== str:
            for j in i:
                if j.isdigit():
                    string=string+j
                if j=='.':
                    string=string+j
                if j=="/":
                    l.append(float(string))
                    string=''
                    break
        else:
            l.append(i)
    return l

sns.boxplot(strip(data))
